    // Carregar dados JSON ao carregar a página
    document.addEventListener('DOMContentLoaded', () => {
      // URL do arquivo JSON
      const url = 'https://ddragon.leagueoflegends.com/cdn/14.10.1/data/en_US/champion.json';

      // Usando fetch para buscar o arquivo
      fetch(url)
        .then(response => {
          // Verifica se a resposta está OK
          if (!response.ok) {
            throw new Error('Erro na rede: ' + response.statusText);
          }
          return response.json(); // Converte a resposta para JSON
        })
        .then(data => {
          // Iterar sobre cada herói no objeto data e logar seus IDs
          Object.keys(data.data).forEach(hero => {
            console.log(hero);
          });
        })
        .catch(error => {
          // Lida com erros de rede ou de conversão
          console.error('Houve um problema com a solicitação fetch:', error);
        });
    });
